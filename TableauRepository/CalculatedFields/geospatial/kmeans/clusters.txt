    set.seed( .arg4[1] )
    lon <- ( .arg1 - mean(.arg1) ) / sd(.arg1)
    lat <- ( .arg2 - mean(.arg2) ) / sd(.arg2)

    dat <- cbind(lon, lat)

    num <- .arg3[1]

    tryCatch({kmeans(dat, num, nstart=num)$cluster}, error = function(e) { 1 })
",     
SUM([Lon]), SUM([Lat]), [Number of Clusters], [Seed]
)
